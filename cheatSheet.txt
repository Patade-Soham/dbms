);

(value1, value2, value3);

FROM table_name
WHERE condition;
SELECT * FROM table_name;
🔥 4. UPDATE Template

SET column = value
WHERE condition;
🔥 5. DELETE Template
DELETE FROM table_name
WHERE condition;

ALTER TABLE table_name DROP COLUMN column_name;

TRUNCATE TABLE table_name;

🔥 8. CONSTRAINT Templates
PRIMARY KEY (column)

FOREIGN KEY (column) REFERENCES table(column)

UNIQUE(column)

CHECK(condition)

DEFAULT value
🔥 9. JOIN Templates
INNER JOIN
SELECT *
FROM table1 t1
INNER JOIN table2 t2
ON t1.col = t2.col;
LEFT JOIN
SELECT *
FROM table1 t1
LEFT JOIN table2 t2
ON t1.col = t2.col;
NATURAL JOIN
SELECT *
FROM table1
NATURAL JOIN table2;
🔥 10. AGGREGATE FUNCTIONS

🔥 11. SUBQUERY Template
SELECT *
FROM table

🔥 12. LIKE Template

🔥 13. TRANSACTION Template
START TRANSACTION;

UPDATE table SET column=value;

SAVEPOINT sp1;

UPDATE table SET column=value;

ROLLBACK TO sp1;

COMMIT;
